# Первоначальная настройка
make init                    # Генерация ключей + запуск базовых сервисов

# Запуск полного стека
make up-cpu                  # Для CPU
make up-gpu                  # Для GPU

# Частичный запуск
make up-base                 # Только Traefik, PostgreSQL, Redis
make up-apps                 # Только приложения
make up-monitoring           # Только мониторинг

# Управление
make down                    # Остановить все
make restart                 # Перезапустить все
make logs                    # Показать логи
make ps                      # Статус контейнеров

# Ollama
make ollama-pull             # Загрузить базовые модели
make ollama-list             # Список моделей

# Утилиты
make health                  # Проверка здоровья
make backup                  # Создать бэкап
make clean                   # Очистка
